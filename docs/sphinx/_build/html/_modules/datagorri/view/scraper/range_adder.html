[docs]    def on_addition(self, func):
        def handle_click(event):
            self.clickSimulate(self._add_btn)
            func(event)
            time.sleep(0.15)
            self.clickSimulateEnd(self._add_btn)
        self._add_btn.bind('<Button-1>', handle_click)
